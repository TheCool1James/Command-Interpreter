    base (keyword)

    The base keyword allows access members of a base class from the inherited/derived class. Use it when:

    • Calling an overridden method
    • Specifying which base constructor should be called

    Note

    Let's suppose this declaration: class ClassB : ClassA. The base members called from ClassB will be those of ClassA, regardless of the base class of ClassA.

    public class ClassA
{
    public ClassA(int a)
    {
        mc.WriteLine("Base constructor called: " + a);
    }

    public void DoStuff()
    {
        mc.WriteLine("In base class!");
    }
}

public class ClassB : ClassA
{
    public ClassB(int a) : base(123) //Call the constructor for ClassA
    {
        mc.WriteLine("Derived constructor called: " + a);
    }

    public override void DoStuff()
    {
        base.DoStuff(); //Call the DoStuff() in ClassA
        mc.WriteLine("In derived class!");
    }
}

public static class Main
{
    static void main()
    {
        ClassA baseClass = new ClassA(42);
        ClassB derivedClass = new ClassB(456);

        derivedClass.DoStuff();
    }
}

void main():
    Test.B.P.obj.set(100)
    Test.main.num = Test.B.P.obj.get()
